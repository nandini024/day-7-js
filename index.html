console.log("Prime numbers from 2 to 50:");

for (let num = 2; num < 50; num++) {
    let flag = true; // Reset flag for each number

    for (let i = 2; i < num; i++) {
        if (num % i == 0) {  
            flag = false;
            break; // Stop checking if a divisor is found
        }
    }

    if (flag) { // If still true, it's a prime number
        console.log(num); // Use this for browser
    }
}

var str = "345678";
var min = str[0];

for (let i of str)
    if (min > i) min = i;

console.log(min, "is the minimum number in the given string");

// Maximum number of given numbers
var max = str[0];

for (let i of str)
    if (max < i) max = i;

console.log(max, "is the maximum number in the given string");
